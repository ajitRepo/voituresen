'#00D46A' : d.status === 'pending' ? '#FF6B00' : '#aaa'; const statusLabel = d.verified ? 'Actif' : d.status === 'pending' ? 'En attente' : d.status || 'Inconnu'; const date = d.created_at ? new Date(d.created_at).toLocaleDateString('fr-FR') : '-'; return ` ${d.name || '-'} ${d.phone || '-'} ${d.city || 'Dakar'} ${statusLabel} ${d.amount ? d.amount.toLocaleString('fr-FR') + ' FCFA' : '-'} ${date} ${!d.verified ? `` : ``} `; }).join(''); } async function activateDealer(id) { try { const r = await fetch('/api/admin/dealers/' + id, { method: 'PATCH', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({verified: true, status: 'active'}) }); const d = await r.json(); if (d.success) { await loadAdminDealers(); alert('Badge activé !'); } } catch(e) { alert('Erreur: ' + e.message); } } async function deactivateDealer(id) { try { const r = await fetch('/api/admin/dealers/' + id, { method: 'PATCH', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({verified: false, status: 'suspended'}) }); const d = await r.json(); if (d.success) { await loadAdminDealers(); alert('Badge désactivé.'); } } catch(e) { alert('Erreur: ' + e.message); } } async function deleteDealer(id) { if (!confirm('Supprimer ce concessionnaire ?')) return; try { await fetch('/api/admin/dealers/' + id, {method: 'DELETE'}); await loadAdminDealers(); } catch(e) { alert('Erreur: ' + e.message); } } // ============ FIN CONCESSIONNAIRES ============
